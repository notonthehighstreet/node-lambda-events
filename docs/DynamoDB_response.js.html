 * Provides a simple interface for responding to DynamoDB Lambda Events
 *
 * @constructor Response
 *
 * @module DynamoDB
 */

  constructor(cb) {
    this.cb = cb;
  }

  respond(status = OK, body = '') {
    if (status === OK) {
      cb(null, body);
